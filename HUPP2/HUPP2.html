def J_pol(alpha): # J_pol med rotation med vinkel alpha.
    return J_proj(-alpha)@np.array([[1, 0],[0, 0]])@J_proj(alpha)

def J_ret(alpha, phi): # J_ret med rotation med vinkel alpha och fasfrskjutning phi.
    return np.matmul(J_proj(-alpha), np.array([[np.exp(1j*phi), 0],[0, 1]]))@J_proj(alpha)

def J_ret2(alpha, delta, ne0, n0, lam):

def I_ut(theta, lam):
    E_in = np.array([1,0])
